# Étape 1 : Build
FROM mcr.microsoft.com/dotnet/sdk:8.0 AS build
WORKDIR /src

# Copier uniquement les fichiers nécessaires pour la restauration (optimise le cache Docker)
COPY ["AxiaLivraisonAPI.csproj", "."]
RUN dotnet restore "AxiaLivraisonAPI.csproj"

# Copier le reste et compiler
COPY . .
RUN dotnet publish -c Release -o /app --no-restore

# Étape 2 : Runtime
FROM mcr.microsoft.com/dotnet/aspnet:8.0
WORKDIR /app
COPY --from=build /app .

# Configurer le port dynamique pour Render
ENV ASPNETCORE_URLS=http://*:$PORT
EXPOSE $PORT

ENTRYPOINT ["dotnet", "AxiaLivraisonAPI.dll"]
